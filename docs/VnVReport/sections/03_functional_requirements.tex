\section{Functional Requirements Evaluation}
\subsection{Area of Testing 1 – User Verification}
\paragraph{Test Case 1 – Valid Sign-Up} \label{test-FR1-1}
\begin{description}
    \item[Initial State:] No existing account in the system. 
    \item[Input:] Valid McMaster email, password, and driver license upload (for drivers).
    \item[Expected Output:] Verification email sent; account activated after confirmation.
    \item[Actual Output:] Verification email sent; account activated after confirmation.
    \item[Result:] Pass
\end{description}

\paragraph{Test Case 2 – Invalid Verification} \label{test-FR1-2}
\begin{description}
    \item[Initial State:] No existing account in the system. 
    \item[Input:] Invalid email format, weak password, or missing driver license (for drivers).
    \item[Expected Output:] Appropriate error messages displayed; account not created.
    \item[Actual Output:] Appropriate error messages displayed; account not created.
    \item[Result:] Pass
\end{description}

\subsection{Area of Testing 3 – Ride Matching}
\paragraph{Test Case 1 – Successful Ride Match} \label{test-FR3-1}
\begin{description}
    \item[Initial State:] Verified users with valid profiles and schedules exist in the system. 
    \item[Input:] A rider and driver with overlapping routes and times.
    \item[Expected Output:] Match proposal generated with a computed cost share presented to both users.
    \item[Actual Output:] Match proposal generated with a computed cost share presented to both users.
    \item[Result:] Pass
\end{description}
\paragraph{Test Case 2 – No Match Found} \label{test-FR3-2}
\begin{description}
    \item[Initial State:] Verified users with valid profiles and schedules exist in the system.
    \item[Input:] A rider and driver with non-overlapping schedules.
    \item[Expected Output:] ``No match found'' message returned to the user.
    \item[Actual Output:]``No match found'' message returned to the user.
    \item[Result:] Pass
\end{description}
\paragraph{Test Case 3 – User Rejection} \label{test-FR3-3}
\begin{description}
    \item[Initial State:] Rider and driver sent a rideshare request.
    \item[Input:] A rider and driver with overlapping schedules and opposite preferences reject the ride.
    \item[Expected Output:] “User rejected” message.
    \item[Actual Output:] “User rejected” message.
    \item[Result:] Pass
\end{description}
