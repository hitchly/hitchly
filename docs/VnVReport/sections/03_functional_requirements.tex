\section{Functional Requirements Evaluation}
\subsection{User Verification}
\begin{enumerate}
    \item test-FR1-1:\label{test-FR1-1}
    Initial State: No existing account in the system. \\
    Input: Valid McMaster email, password, and driver license upload (for drivers).\\
    Expected Output: Verification email sent; account activated after confirmation.\\
    Actual Output:Verification email sent; account activated after confirmation.\\
    Result:Pass

    \item test-FR1-2:\label{test-FR1-2}
    Initial State: No existing account in the system. \\
    Input: Non-McMaster email or invalid license format.\\
    Expected Output: Account creation rejected with an error message displayed to the user.\\
    Actual Output:Account creation rejected with an error message displayed to the user.\\
    Result:Pass
\end{enumerate}

\subsection{Ride Matching}
\begin{enumerate}
    \item test-FR2-1:\label{test-FR2-1}
    Initial State: Verified users with valid profiles and schedules exist in the system. \\
    Input: A rider and driver with overlapping routes and times. \\
    Expected Output: Match proposal generated with a computed cost share presented to both users. \\
    Actual Output: Match proposal generated with a computed cost share presented to both users. \\
    Result:Pass

    \item test-FR2-2:\label{test-FR2-2}
    Initial State: Verified users with valid profiles and schedules exist in the system.
    Input: A rider and driver with non-overlapping schedules. \\
    Expected Output: ``No match found'' message returned to the user.\\
    Actual Output:``No match found'' message returned to the user.\\
    Result: Pass

    \item test-FR2-3:\label{test-FR2-3}
    Initial State: Rider and driver sent a rideshare request.
    Input: A rider and driver with overlapping schedules and opposite preferences reject the ride. \\
    Expected Output: “User rejected” message.\\
    Actual Output: “User rejected” message.\\
    Result: Pass
\end{enumerate}
